# 链表

```javascript
function ListNode(val, next) {
  this.val = val === undefined ? 0 : val;
  this.next = next === undefined ? null : next;
}
```

## 删除操作

### 删除链表的倒数第 N 个结点

```javascript
// 快慢指针
var removeNthFromEnd = function (head, n) {
  let p = head;
  let index = 1;
  // 缓存指针,并计算长度
  let map = {};
  while (p) {
    map[index++] = p;
    p = p.next;
  }
  // 链表长度
  let len = index--;
  // 删除的索引
  let removeIndex = len - n;
  // 删除的对象
  let removeObj = map[removeIndex];
  // 删除的对象上一个
  let prevObj = map[removeIndex - 1];
  // 删除的对象下一个
  let nextObj = removeObj.next;

  if (prevObj == null) {
    if (nextObj == null) {
      return null;
    } else {
      return nextObj;
    }
  } else {
    prevObj.next = nextObj;
    return head;
  }
};
```

### 删除重复元素

输入 head = [1,2,3,3,4,4,5] 输出：[1,2,5]

```javascript
// 伪节点法
var deleteDuplicates = function (head) {
  if (head) {
    let newHead = new ListNode("a");
    newHead.next = head;
    let temp = newHead.next;
    let prevNode = newHead;
    let repeatNum = null;
    while (temp) {
      //1 检查当前和 repeatNum
      if (repeatNum !== null) {
        if (temp.next) {
          if (temp.next.val === repeatNum) {
            prevNode.next = temp.next;
            temp = temp.next;
          } else {
            prevNode.next = temp.next;
            temp = temp.next;
            repeatNum = null;
          }
        } else {
          prevNode.next = temp.next;
          repeatNum = null;
        }
      } else {
        // 判断当前节点和后边节点值
        if (!temp.next) {
          temp = temp.next;
        } else {
          if (temp.val === temp.next.val) {
            repeatNum = temp.val;
            temp.next = temp.next.next;
          } else {
            prevNode = temp;
            temp = temp.next;
          }
        }
      }
    }
    return newHead.next;
  } else {
    return head;
  }
};
```

## 合并操作

### 链表归并排序

```javascript
const merge = (head1, head2) => {
  const dummyHead = new ListNode(0);
  let temp = dummyHead,
    temp1 = head1,
    temp2 = head2;
  while (temp1 !== null && temp2 !== null) {
    if (temp1.val <= temp2.val) {
      temp.next = temp1;
      temp1 = temp1.next;
    } else {
      temp.next = temp2;
      temp2 = temp2.next;
    }
    temp = temp.next;
  }
  if (temp1 !== null) {
    temp.next = temp1;
  } else if (temp2 !== null) {
    temp.next = temp2;
  }
  return dummyHead.next;
};

const toSortList = (head, tail) => {
  if (head === null) {
    return head;
  }
  if (head.next === tail) {
    head.next = null;
    return head;
  }
  let slow = head,
    fast = head;
  while (fast !== tail) {
    slow = slow.next;
    fast = fast.next;
    if (fast !== tail) {
      fast = fast.next;
    }
  }
  const mid = slow;
  return merge(toSortList(head, mid), toSortList(mid, tail));
};

var sortList = function (head) {
  return toSortList(head, null);
};
```

## 反转链表
头插法
```javascript

var reverseList = function (head) {
  let result = null;
  let current = head;
  while (current) {
    const nextNodes = current.next;
    current.next = result;
    result = current;
    current = nextNodes;
  }
  return result;
};
```
