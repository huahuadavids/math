### 反转链表
```c
struct ListNode* reverseList(struct ListNode* head){
    if(head == NULL){
        return head;
    }
    struct ListNode* newHead = NULL;
    struct ListNode* tmp = NULL;
    while(head != NULL){
        tmp = head->next;
        head->next = newHead;
        newHead = head;
        head =tmp;
    }
    return newHead;
}
```
