## 亦或

// 特性 
// 亦或操作和顺序无关,因为一堆数的运算结果和这些数一起的1的个数有关
n ^ 0 = n
n ^ n = 0;
a ^ b = b ^ a;
a ^ b ^ c = b ^ c ^ a;

// 交换 a 和 b, 如果是一个数组中的a和b表示相同的下标，那么会被抹为0
a = a ^ b;
b = a ^ b;
a = a ^ b;
